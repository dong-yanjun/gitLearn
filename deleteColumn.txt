Private Sub CommandButton1_Click()

Dim count As Integer
Dim columnNum As Integer

Dim value As String

count = Cells(Rows.count, 1).End(xlUp).Row

columnNum = Cells(3, Columns.count).End(xlToLeft).Column

For i = 3 To count

If (Cells(i, 11).MergeCells) Then

value = "C" & i & ":E" & i & "," & "H" & i & ":H" & i & "," & "L" & i & ":O" & i

Range(value).Select

Selection.Delete shift:=xlToLeft

End If

Next i
End Sub
